async function togglePlatform(platform) {
  if (!platform.connected) {
    // Facebook and Instagram both use Meta OAuth
    if (platform.name === 'Facebook' || platform.name === 'Instagram') {
      const data = await api.get('/connections/meta/auth')
      if (data?.authUrl) {
        const popup = window.open(data.authUrl, 'meta-oauth', 'width=600,height=700')
        const timer = setInterval(async () => {
          if (popup?.closed) {
            clearInterval(timer)
            // Reload both FB and IG status after popup closes
            await refreshMetaConnections()
          }
        }, 500)
      }
      return
    }
  } else {
    await api.delete(`/connections/${platform.name.toLowerCase()}`)
    platform.connected = false
    platform.handle = null
    platform.stats = []
  }
}

async function refreshMetaConnections() {
  const data = await api.get('/connections')
  if (data?.connections) {
    data.connections.forEach(conn => {
      const p = socialPlatforms.value.find(x => x.name === conn.platform)
      if (p) {
        p.connected = conn.connected
        p.handle    = conn.handle || null
        p.stats     = conn.stats  || []
      }
    })
  }
}
